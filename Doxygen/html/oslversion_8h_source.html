34 // Versions. There are three different sets of version numbers:
35 // (a) The version of the OSL language specification itself, i.e.,
36 // what is OSL syntax, semantics, and standard library.
37 // (b) The version of the OSL library, i.e., this very software whose
38 // source code you are reading.
39 // (c) The version of the .oso file format that this library uses
40 // for compiled shaders.
41 // These are all independent, though loosely coupled.
42 
43 // Version of this library:
44 #define OSL_LIBRARY_VERSION_MAJOR 1
45 #define OSL_LIBRARY_VERSION_MINOR 6
46 #define OSL_LIBRARY_VERSION_PATCH 2
47 #define OSL_LIBRARY_VERSION_RELEASE_TYPE dev
48 
49 #define OSL_LIBRARY_VERSION_CODE (10000 * OSL_LIBRARY_VERSION_MAJOR + \
50  100 * OSL_LIBRARY_VERSION_MINOR + \
51  OSL_LIBRARY_VERSION_PATCH)

53 // Magic macros to make OSL_LIBRARY_VERSION_STRING that looks like "1.2.3"
54 #define OSL_MAKE_VERSION_STRING2(a,b,c,d) #a "." #b "." #c #d
55 #define OSL_MAKE_VERSION_STRING(a,b,c,d) OSL_MAKE_VERSION_STRING2(a,b,c,d)
56 #define OSL_LIBRARY_VERSION_STRING \
57  OSL_MAKE_VERSION_STRING(OSL_LIBRARY_VERSION_MAJOR, \
58  OSL_LIBRARY_VERSION_MINOR, \
59  OSL_LIBRARY_VERSION_PATCH, \
60  OSL_LIBRARY_VERSION_RELEASE_TYPE)
61 #define OSL_INTRO_STRING "OpenShadingLanguage " OSL_LIBRARY_VERSION_STRING

65 // Version numbers for the compiled shader format. The major number
66 // should only be changed if old .oso files should not be expected to
67 // work with current versions of the library (this should be EXTREMELY
68 // rare, hopefully never). The minor number may be changed if newly
69 // compiled shaders won't work for old libraries, which sometimes
70 // happens but still should not be often. Adding a new hint or a new
71 // shader instruction is not considered a change to the file format!
72 #define OSO_FILE_VERSION_MAJOR 1
73 #define OSO_FILE_VERSION_MINOR 0
