80 static __device__ __inline__ void mapToDisk( optix::float2& sample )
81 {
82  float phi, r;
83  float a = 2.0f * sample.x - 1.0f; // (a,b) is now on [-1,1]^2
84  float b = 2.0f * sample.y - 1.0f; //
85  if (a > -b) { // reg 1 or 2
86  if (a > b) { // reg 1, also |a| > |b|
87  r = a;
88  phi = (M_PIf/4.0f) * (b/a);
89  } else { // reg 2, also |b| > |a|
90  r = b;
91  phi = (M_PIf/4.0f) * (2.0f - a/b);
92  }
93  } else { // reg 3 or 4
94  if (a < b) { // reg 3, also |a|>=|b| && a!=0
95  r = -a;
96  phi = (M_PIf/4.0f) * (4.0f + b/a);
97  } else { // region 4, |b| >= |a|, but
98  // a==0 and b==0 could occur.
99  r = -b;
100  phi = b != 0.0f ? (M_PIf/4.0f) * (6.0f - a/b) :
101  0.0f;
102  }
103  }
104  float u = r * cosf( phi );
105  float v = r * sinf( phi );
106  sample.x = u;
107  sample.y = v;
108 }

111 // Uniformly sample the surface of a Parallelogram. Return probability
112 // of the given sample
113 static
114 __device__ __inline__ void sampleParallelogram( const ParallelogramLight& light,
115  const optix::float3& hit_point,
116  const optix::float2& sample,
117  optix::float3& w_in,
118  float& dist,
119  float& pdf )
120 {
121  using namespace optix;
122 
123  float3 on_light = light.corner + sample.x*light.v1 + sample.y*light.v2;
124  w_in = on_light - hit_point;
125  float dist2 = dot( w_in, w_in );
126  dist = sqrt( dist2 );
127  w_in /= dist;
128 
129  float3 normal = cross( light.v1, light.v2 );
130  float area = length( normal );
131  normal /= area;
132  float cosine = -dot( normal, w_in );
133  pdf = dist2 / ( area * cosine );
134 }

137 // Create ONB from normal. Resulting W is Parallel to normal
138 static
139 __device__ __inline__ void createONB( const optix::float3& n,
140  optix::float3& U,
141  optix::float3& V,
142  optix::float3& W )
143 {
144  using namespace optix;
145 
146  W = normalize( n );
147  U = cross( W, make_float3( 0.0f, 1.0f, 0.0f ) );
148  if ( fabsf( U.x) < 0.001f && fabsf( U.y ) < 0.001f && fabsf( U.z ) < 0.001f )
149  U = cross( W, make_float3( 1.0f, 0.0f, 0.0f ) );
150  U = normalize( U );
151  V = cross( W, U );
152 }

155 static
156 __device__ __inline__ void createONB( const optix::float3& n,
157  optix::float3& U,
158  optix::float3& V)
159 {
160  using namespace optix;
161 
162  U = cross( n, make_float3( 0.0f, 1.0f, 0.0f ) );
163  if ( dot(U, U) < 1.e-3f )
164  U = cross( n, make_float3( 1.0f, 0.0f, 0.0f ) );
165  U = normalize( U );
166  V = cross( n, U );
167 }

169 // sample hemisphere with cosine density
170 static
171 __device__ __inline__ void sampleUnitHemisphere( const optix::float2& sample,
172  const optix::float3& U,
173  const optix::float3& V,
174  const optix::float3& W,
175  optix::float3& point )
176 {
177  using namespace optix;
178 
179  float phi = 2.0f * M_PIf*sample.x;
180  float r = sqrt( sample.y );
181  float x = r * cos(phi);
182  float y = r * sin(phi);
183  float z = 1.0f - x*x -y*y;
184  z = z > 0.0f ? sqrt(z) : 0.0f;
185 
186  point = x*U + y*V + z*W;
187 }
