60 static
61 __device__ void phongShade( float3 p_Kd,
62  float3 p_Ka,
63  float3 p_Ks,
64  float3 p_normal,
65  float p_phong_exp,
66  float3 p_reflectivity )
67 {
68  float3 hit_point = ray.origin + t_hit * ray.direction;
69 
70  // ambient contribution
71 
72  float3 result = p_Ka * ambient_light_color;
73 
74  // compute direct lighting
75  unsigned int num_lights = lights.size();
76  for(int i = 0; i < num_lights; ++i) {
77  BasicLight light = lights[i];
78  float Ldist = optix::length(light.pos - hit_point);
79  float3 L = optix::normalize(light.pos - hit_point);
80  float nDl = optix::dot( p_normal, L);
81 
82  // cast shadow ray
83  float3 light_attenuation = make_float3(static_cast<float>( nDl > 0.0f ));
84  if ( nDl > 0.0f && light.casts_shadow ) {
85  PerRayData_shadow shadow_prd;
86  shadow_prd.attenuation = make_float3(1.0f);
87  optix::Ray shadow_ray = optix::make_Ray( hit_point, L, shadow_ray_type, scene_epsilon, Ldist );
88  rtTrace(top_shadower, shadow_ray, shadow_prd);
89  light_attenuation = shadow_prd.attenuation;
90  }
91 
92  // If not completely shadowed, light the hit point
93  if( fmaxf(light_attenuation) > 0.0f ) {
94  float3 Lc = light.color * light_attenuation;
95 
96  result += p_Kd * nDl * Lc;
97 
98  float3 H = optix::normalize(L - ray.direction);
99  float nDh = optix::dot( p_normal, H );
100  if(nDh > 0) {
101  float power = pow(nDh, p_phong_exp);
102  result += p_Ks * power * Lc;
103  }
104  }
105  }
106 
107  if( fmaxf( p_reflectivity ) > 0 ) {
108 
109  // ray tree attenuation
110  PerRayData_radiance new_prd;
111  new_prd.importance = prd.importance * optix::luminance( p_reflectivity );
112  new_prd.depth = prd.depth + 1;
113 
114  // reflection ray
115  if( new_prd.importance >= 0.01f && new_prd.depth <= max_depth) {
116  float3 R = optix::reflect( ray.direction, p_normal );
117  optix::Ray refl_ray = optix::make_Ray( hit_point, R, radiance_ray_type, scene_epsilon, RT_DEFAULT_MAX );
118  rtTrace(top_object, refl_ray, new_prd);
119  result += p_reflectivity * new_prd.result;
120  }
121  }
122 
123  // pass the color back up the tree
124  prd.result = result;
125 }
